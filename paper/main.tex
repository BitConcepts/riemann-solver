\begin{abstract}
We certify strict log-concavity of the Riemann--Jacobi kernel~$\Phi(u)$
appearing in the Fourier cosine representation
\[
\Xi(t) = \int_0^\infty \Phi(u)\cos(tu)\,du
\]
of the Riemann Xi function.
The proof combines an elementary algebraic analysis of the dominant term,
rigorous interval arithmetic on~$[0,1]$, a no-cancellation certification
of $(\log\Phi)'' < 0$ on~$[1,3]$, and an explicit monotonicity tail bound for~$u\geq 3$.

We also audit the commonly cited P\'{o}lya-type bridge from log-concavity of a
kernel to real-rootedness of its Fourier transform.
Direct inspection of P\'{o}lya (1927) Satz~I and Satz~II shows that these theorems
are, respectively, a preservation theorem and a Mellin--Fourier theorem; neither states
the log-concavity criterion needed to derive the Riemann Hypothesis from the result
proved here.
Accordingly, \textbf{no unconditional RH claim is made.}

The unconditional contribution of this paper is the certified strict log-concavity
of~$\Phi$ on~$[0,\infty)$.
The possible implication to RH is recorded only as a conditional consequence of
an unsourced P\'{o}lya-type criterion (Hypothetical Criterion~\ref{hcrit:polya_star}).
The proof of log-concavity is independent of the P\'{o}lya-source audit.

\section{Introduction}
\label{sec:intro}

The Riemann Hypothesis (RH), posed by Bernhard Riemann in 1859, asserts that
all nontrivial zeros of the Riemann zeta function~$\zeta(s)$ lie on the
critical line~$\operatorname{Re}(s) = 1/2$.
Equivalently, all zeros of the Riemann Xi function
\begin{equation}\label{eq:xi}
\Xi(t) := \xi\!\left(\tfrac{1}{2} + it\right), \quad
\xi(s) := \tfrac{1}{2}s(s-1)\pi^{-s/2}\Gamma(s/2)\zeta(s),
\end{equation}
are real.
It is well known~\cite{titchmarsh1986} that~$\Xi(t)$ admits the Fourier cosine
representation
\begin{equation}\label{eq:fourier}
\Xi(t) = \int_0^\infty \Phi(u)\cos(tu)\,du,
\end{equation}
where the \emph{Riemann--Jacobi kernel} is
\begin{equation}\label{eq:phi}
\Phi(u) = 4\sum_{n=1}^{\infty}\varphi_n(u), \quad
\varphi_n(u) = \bigl(2\pi^2 n^4 e^{9u/2} - 3\pi n^2 e^{5u/2}\bigr)e^{-\pi n^2 e^{2u}}.
\end{equation}
We verify computationally that
$\int_0^\infty \Phi(u)\,du = \xi(1/2) \approx 0.4971$ to 15~digits.

\paragraph{Approach.}
This paper proves a structural property of the Riemann--Jacobi kernel:
strict log-concavity of~$\Phi$ on~$[0,\infty)$ (Theorem~\ref{thm:main}).
Because the Fourier transform of this kernel is~$2\Xi$, this property is
naturally relevant to the real-zero problem for~$\Xi$.
However, the implication from kernel log-concavity to real-rootedness of the
Fourier transform is not established by the P\'{o}lya sources audited here.
We therefore separate the unconditional log-concavity theorem from the conditional
RH implication.

The paper should be read as a certification of log-concavity, not as a proof of RH.
The RH consequence appears only in a conditional remark
(Remark~\ref{thm:rh_conditional}), dependent on a hypothetical criterion not
verified in the primary source trail.
The source audit (Section~\ref{sec:polya_audit_main}) is not needed for the proof
of Theorem~\ref{thm:main}; it only governs the interpretation of possible RH
consequences.

\paragraph{Prior work.}
The log-concavity of the dominant term~$\varphi_1$ was observed by Csordas and
Varga~\cite{csordas1989}.
The Tur\'{a}n inequalities for the Taylor coefficients of~$\Xi$ were established
by Csordas, Norfolk, and Varga~\cite{csordas1986}, and higher-order generalizations
by Dimitrov and Lucas~\cite{dimitrov2011} and Griffin, Ono, Rolen, and
Zagier~\cite{griffin2019}.
The connection between log-concavity and the de~Bruijn--Newman constant was explored
by de~Bruijn~\cite{debruijn1950} and Rodgers and Tao~\cite{rodgers2020}.
The key distinction of this work is verifying log-concavity of the
\emph{full kernel}~$\Phi$, not just the dominant term.

\paragraph{Claims intentionally not made.}
This paper does not claim:
\begin{enumerate}[leftmargin=2em,itemsep=1pt]
\item that RH is proved;
\item that P\'{o}lya (1927) Satz~I or Satz~II implies the log-concavity criterion;
\item that Csordas--Varga (1989) Theorem~2.2 states the log-concavity criterion;
\item that Lean verifies RH;
\item that the interval arithmetic certificates are formalized in Lean;
\item that Hypothetical Criterion~\ref{hcrit:polya_star} is established in the
peer-reviewed literature.
\end{enumerate}

\section{Properties of $\Phi$}
\label{sec:phi}

\begin{proposition}\label{prop:phi_properties}
The kernel~$\Phi$ defined by~\eqref{eq:phi} satisfies:
\begin{enumerate}[label=(\roman*)]
\item $\Phi(u) > 0$ for all~$u \geq 0$;
\item $\Phi(-u) = \Phi(u)$ for all~$u$;
\item $\Phi \in L^1(\mathbb{R})$;
\item $\Phi(u) = O\bigl(e^{-\pi e^{2u}}\bigr)$ as~$u \to +\infty$;
\item $\Phi$ is real analytic on~$\mathbb{R}$.
\end{enumerate}
\end{proposition}

\begin{proof}
Properties~(i)--(iii) are classical; see~\cite[\S2.10]{titchmarsh1986}
and~\cite[Theorem~A]{csordas1989}.
Property~(iv) follows from the dominant exponential factor $e^{-\pi e^{2u}}$ in~$\varphi_1$.
For~(v): fix any $u_0\in\mathbb{R}$.
For $r = \pi/8$, any $z$ with $|z - u_0| < r$ satisfies
$\operatorname{Re}(e^{2z}) \geq e^{2(u_0-r)}/\sqrt{2} =: c > 0$.
Thus $|\varphi_n(z)| \leq A\,n^4\,e^{-\pi c n^2/2}$ uniformly on $|z-u_0|\leq r/2$.
Since $\sum n^4 e^{-\pi c n^2/2} < \infty$, the Weierstrass $M$-test gives uniform
convergence; hence $\Phi$ is real analytic at $u_0$.
($\Phi$ is \emph{not} entire; only real analyticity on~$\mathbb{R}$, as required by
condition~(v) of Hypothetical Criterion~\ref{hcrit:polya_star}, is claimed.)

\emph{Numerical verification.}
$\Phi(u) > 0$ at 10{,}001 uniformly spaced points on $[0,1]$
($\min = 5.51 \times 10^{-7}$ at $u = 1$);
$|\Phi(u) - \Phi(-u)|/|\Phi(u)| < 10^{-70}$ at 8~test points.
\end{proof}

\section{Algebraic Core}
\label{sec:algebraic}

\begin{theorem}[Algebraic Core]\label{thm:algebraic}
$(\log\varphi_1)''(u) < 0$ for all~$u \geq 0$.
\end{theorem}

\begin{proof}
Write $\varphi_1 = g \cdot E$ where $g(u) = \pi e^{5u/2}h(u)$,
$h(u) = 2\pi e^{2u} - 3$, and $E(u) = e^{-\pi e^{2u}}$.
Since $h(u) \geq 2\pi - 3 > 0$ for~$u \geq 0$, we have $\varphi_1 > 0$. Then
\[
\log\varphi_1 = \log\pi + \tfrac{5}{2}u + \log h(u) - \pi e^{2u},
\quad
(\log\varphi_1)'' = (\log h)'' - 4\pi e^{2u}.
\]
With $h' = 4\pi e^{2u}$ and $h'' = 8\pi e^{2u}$:
\[
(\log h)'' = \frac{h''h - (h')^2}{h^2} = \frac{-24\pi e^{2u}}{h(u)^2} < 0.
\]
Both terms are negative, so $(\log\varphi_1)'' < 0$.
\end{proof}

\begin{remark}
$(\log\varphi_1)''(0) \approx -19.56$, monotonically decreasing.
The series for~$\Phi''(0)$ converges to $-68.052\ldots$ by $N = 5$ terms.
\end{remark}

\section{Tail Estimate}
\label{sec:tail}

\begin{lemma}\label{lem:tail_decay}
For $n \geq 2$ and $u \geq 0$: $e^{-\pi n^2 e^{2u}} \leq e^{-3\pi}\cdot e^{-\pi e^{2u}}$.
\end{lemma}

\begin{proof}
Equivalent to $(n^2 - 1)e^{2u} \geq 3$, holding since $n^2 - 1 \geq 3$ and $e^{2u} \geq 1$.
\end{proof}

\begin{proposition}\label{prop:tail_ratio}
For $u \geq 0$: $|R(u)|/\varphi_1(u) < 1/50$, where $R = \sum_{n\geq 2}\varphi_n$.
\end{proposition}

\begin{proof}
$|\varphi_n|/\varphi_1 \leq 2n^4 e^{-\pi(n^2-1)e^{2u}}$ (since $B_n(u)/n^4 < 2$).
At $u = 0$: $\sum_{n\geq 2} 2n^4 e^{-\pi(n^2-1)} < 0.006 < 1/50$.
\end{proof}

\begin{remark}\label{rem:truncation}
The interval arithmetic retains $n = 1,\ldots,5$ and omits
$\delta := 4\sum_{n \geq 6}\varphi_n$.
Truncation errors on~$[0,1]$:
$|\delta| \leq 7.03 \times 10^{-43}$,
$|\delta'| \leq 1.18 \times 10^{-39}$,
$|\delta''| \leq 1.98 \times 10^{-36}$,
certified by interval arithmetic.
At $u = 1$, the dominant cross term $|\delta''\cdot\Phi_5| \approx 1.09 \times 10^{-42}$
gives total error $\leq 1.15 \times 10^{-42}$.
Since $|Q_{\Phi_5}| \geq 3.36 \times 10^{-12}$, the safety factor exceeds $2.9 \times 10^{30}$.
\end{remark}

\section{Interval Arithmetic Verification}
\label{sec:ia}

\begin{definition}
The \emph{log-concavity numerator} of a positive function~$f$ is
$Q_f(u) := f''(u)f(u) - f'(u)^2$.
\end{definition}

\begin{theorem}\label{thm:ia}
$Q_\Phi(u) < 0$ for all~$u \in [0, 1]$.
\end{theorem}

\begin{proof}
Closed-form derivatives:
\begin{align*}
g_n' &= 9\pi^2 n^4 e^{9u/2} - \tfrac{15}{2}\pi n^2 e^{5u/2}, &
g_n'' &= \tfrac{81}{2}\pi^2 n^4 e^{9u/2} - \tfrac{75}{4}\pi n^2 e^{5u/2}, \\
E_n' &= -2\pi n^2 e^{2u}\cdot E_n, &
E_n'' &= \bigl(-4\pi n^2 e^{2u} + 4\pi^2 n^4 e^{4u}\bigr)\cdot E_n.
\end{align*}
All computations use \texttt{mpmath.iv} at 60-digit precision, retaining $n = 1,\ldots,5$.
Partition $[0, 0.949]$ into 1{,}898 subintervals and $[0.949, 1.0]$ into 51{,}000;
on each, evaluate~$Q_{\Phi_5}$ via interval arithmetic and certify the upper bound
is negative.

\textbf{Result:} All 52{,}898 subintervals certified $Q_{\Phi_5}(u) < 0$, maximum upper bound
$-3.36 \times 10^{-12}$.
Since truncation error $\leq 1.15 \times 10^{-42}$ (Remark~\ref{rem:truncation}),
$Q_\Phi(u) < 0$ on~$[0,1]$.
Cross-validation at 80-digit precision at 10~selected points: all within enclosures.
\end{proof}

\section{Extended Certification on $[1.0, 3.0]$}
\label{sec:extended}

Direct interval arithmetic on $Q_\Phi$ fails for $u > 1$ due to catastrophic
cancellation ($40\times$).
We certify $(\log\Phi)''(u) < 0$ directly.

\begin{lemma}[Uniform tail bound]\label{lem:wtail}
For all $u \geq 1$:
$|W_{\mathrm{tail}}(u)| \leq 2\times 10^{-25}$,
where $W_{\mathrm{tail}}(u) := (\log\Phi)''(u) - (\log\varphi_1)''(u)$.

Complete derivation in Appendix~\ref{sec:appE}.
\end{lemma}

\begin{theorem}\label{thm:extended}
$(\log\Phi)''(u) < 0$ for all~$u \in [1.0, 3.0]$.

All proof-critical estimates used in this certification are derived in
Appendix~\ref{sec:appE}; the scripts are reproducibility checks and certify
numerical constants, not substitutes for the written proof.
\end{theorem}

\begin{proof}
Decompose $(\log\Phi)'' = W_1 + W_{\mathrm{tail}}$ where
$W_1(u) = -24\pi e^{2u}/h(u)^2 - 4\pi e^{2u}$ is proved negative in
Theorem~\ref{thm:algebraic}.
For each $i = 0,\ldots,100$, the verifier
(\texttt{proof/verify\_ia\_1\_to\_3.py}, Appendix~\ref{sec:appE})
certifies by interval arithmetic that
$\operatorname{upper}(W_1(I_i) + B(I_i)) < 0$
over $I_i = [u_i-0.01,\, u_i+0.01]$, where $B(u)$ is the
interval-arithmetic upper bound on $|W_{\mathrm{tail}}(u)|$
(Appendix~\ref{sec:appE}, Lemma~E.6).
Adjacent intervals overlap; their union $[0.99,3.01]\supset[1.0,3.0]$.

\textbf{Result:} All 101 checks certified; maximum $B \leq 1.12\times 10^{-23}$;
minimum margin $(-W_1 - B) = 91.3$ at $u=1.0$.
Since $|W_{\mathrm{tail}}|\leq B$:
$(\log\Phi)'' = W_1 + W_{\mathrm{tail}} \leq W_1 + B < 0$.
\end{proof}

\section{Tail Bound for $u \geq 3.0$}
\label{sec:tailbound}

\begin{theorem}\label{thm:perturbation_tail}
$(\log\Phi)''(u) < 0$ for all~$u \geq 3.0$.
\end{theorem}

\begin{proof}
At $u = 3$:
$W_1(3) < -4\pi e^6 < -1000$ and
$\varepsilon^*(3) \leq 32\,e^{-3\pi e^6} < 10^{-1636}$,
so $W_1(3) + 204\,\varepsilon^*(3) < 0$.

For $u > 3$: $W_1'(u) < 0$ holds since $h^3 - 6h - 36 > 0$ for $h \geq 3.92$
(a fortiori for $u \geq 3$); each term of $\varepsilon^*$ is strictly decreasing
term-by-term (Lemma~\ref{lem:epsstar}, Appendix~\ref{sec:appE}).
Since both $W_1$ and $204\varepsilon^*$ are strictly decreasing,
$W_1(u) + 204\,\varepsilon^*(u) \leq W_1(3) + 204\,\varepsilon^*(3) < 0$.

Table: $\varepsilon^*(1.0) \approx 1.82\times 10^{-29}$,
$\varepsilon^*(1.5) \approx 1.96\times 10^{-81}$,
$\varepsilon^*(2.0) \approx 1.07\times 10^{-222}$,
$\varepsilon^*(3.0) < 10^{-1636}$.
\end{proof}

\section{Main Result}
\label{sec:main}

\begin{theorem}[Log-concavity of the Riemann--Jacobi kernel]\label{thm:main}
The kernel~$\Phi$ defined by~\eqref{eq:phi} satisfies $(\log\Phi)''(u) < 0$
for all~$u \geq 0$.
\end{theorem}

\begin{proof}
Combine Theorems~\ref{thm:ia} ($u \in [0,1]$), \ref{thm:extended} ($u \in [1.0, 3.0]$),
and~\ref{thm:perturbation_tail} ($u \geq 3.0$).
\end{proof}

\section{Source-Critical Audit of P\'{o}lya (1927)}
\label{sec:polya_audit_main}

The theorem bridge between log-concavity of~$\Phi$ and real-rootedness of~$\Xi$ is
attributed in the literature to P\'{o}lya (1927)~\cite{polya1927}.
This section is \textbf{not part of the proof} of Theorem~\ref{thm:main}.
It audits whether the commonly cited P\'{o}lya sources support the log-concavity
criterion, and establishes the source-dependency status for the conditional
RH consequence.

\medskip
\noindent\textbf{Source.}
G.\ P\'{o}lya, ``\"{U}ber trigonometrische Integrale mit nur reellen Nullstellen,''
\textit{J.\ reine angew.\ Math.}\ \textbf{158} (1927), pp.~6--18.
Digitisation: Niedersächsische Staats- und Universitätsbibliothek Göttingen (GDZ),
LOG\,Id: LOG\_0006.
Literal title translation: ``On trigonometric integrals with only real zeros.''

\subsection{Exact German text of Satz~I (p.~7)}\label{sec:satzi}

\begin{quote}\itshape
I.\ Die Funktion $\varphi(t)$ sei analytisch f\"{u}r alle reellen Werte von $t$
und von folgender Beschaffenheit: Wenn das Integral
\[\int_{-\infty}^{+\infty} F(t)\,e^{izt}\,dt\]
nur reelle Nullstellen hat, so hat (wie auch $F(t)$ sonst beschaffen sei) das Integral
\[\int_{-\infty}^{+\infty} \varphi(t)\,F(t)\,e^{izt}\,dt\]
ebenfalls nur reelle Nullstellen.  Hierzu ist notwendig und hinreichend,
da\ss{} $C\,e^{\gamma^{2}}\varphi(it)$ eine ganze Funktion vom Geschlecht~1 sei,
die nur reelle Nullstellen besitzt und f\"{u}r reelles $t$ reelle Werte annimmt;
$C$ bedeutet eine nichtverschwindende, $\gamma$ eine reelle, nichtnegative Konstante.
\end{quote}

\paragraph{Literal English translation of Satz~I.}
\begin{quote}
I.\ The function $\varphi(t)$ shall be analytic for all real values of~$t$ and of
the following nature: if the integral
$\int_{-\infty}^{+\infty} F(t)\,e^{izt}\,dt$ has only real zeros,
then (whatever the nature of~$F(t)$ otherwise) the integral
$\int_{-\infty}^{+\infty} \varphi(t)\,F(t)\,e^{izt}\,dt$ also has only real zeros.
For this, the necessary and sufficient condition is that
$C\,e^{\gamma^{2}}\varphi(it)$ be an entire function of genus~$1$
possessing only real zeros and taking real values for real~$t$;
$C$ denotes a non-vanishing constant, $\gamma$ a real, non-negative constant.
\end{quote}

\paragraph{Assessment.}
In modern terms, $\varphi$ preserves real-rootedness of Fourier transforms if and only if
$\varphi$ belongs to the Laguerre--P\'{o}lya class~$\mathcal{LP}$.
Satz~I is a \textbf{preservation theorem}: it requires the Fourier transform of the
input kernel to already have only real zeros---which is precisely the conclusion
sought for~$\Xi$---and so cannot establish real-rootedness from scratch.

\subsection{Exact German text of Satz~II (pp.~7--8)}\label{sec:satzii}

\begin{quote}\itshape
II.\ Es sei die Funktion $f(t)$ f\"{u}r positive Werte von $t$ definiert,
reellwertig, absolut integrabel und f\"{u}r gen\"{u}gend gro\ss{}es $t$
der Ungleichung
\[|f(t)| < B\,e^{-t^{1+\beta}}\]
unterworfen ($B$, $\beta$ sind positive Konstanten).
Ferner sei $f(t)$ in einer Umgebung des Punktes $t = 0$ analytisch.
Dann stellt das Integral
\[\int_0^{\infty} t^{z-1}\,f(t)\,dt = H(z)\]
eine meromorphe Funktion dar.
Wenn die Funktion $H(z)$ nur reelle negative Nullstellen besitzt,
und $q$ eine positive ganze Zahl ist, so besitzt die durch das Integral
\[\int_{-\infty}^{+\infty} f(t^{2q})\,e^{izt}\,dt\]
dargestellte ganze Funktion nur reelle Nullstellen.
\end{quote}

\paragraph{Literal English translation of Satz~II.}
\begin{quote}
II.\ Let the function $f(t)$ be defined for positive values of~$t$, real-valued,
absolutely integrable, and for sufficiently large~$t$ subject to the inequality
$|f(t)| < B\,e^{-t^{1+\beta}}$ ($B$,~$\beta$ positive constants).
Furthermore let $f(t)$ be analytic in a neighbourhood of the point $t = 0$.
Then the integral $\int_0^{\infty} t^{z-1} f(t)\,dt = H(z)$ represents a
meromorphic function.
If the function $H(z)$ has only real negative zeros, and $q$ is a positive
integer, then the entire function represented by the integral
$\int_{-\infty}^{+\infty} f(t^{2q})\,e^{izt}\,dt$ has only real zeros.
\end{quote}

\paragraph{Assessment.}
The key hypothesis is that $H(z)$ has only real negative zeros---a condition on the
\textbf{Mellin transform} $H(z)=\int_0^\infty t^{z-1}f(t)\,dt$, not a log-concavity
condition on~$f$.
Log-concavity of~$f$ does not appear anywhere in Satz~II.
The conclusion concerns the Fourier integral of $f(t^{2q})$, not of~$f$ directly.

\subsection{Audit scope and verdict}\label{sec:verdict}

\begin{center}

}}

\subsection{Extended source analysis}\label{sec:extended_source}

The following primary sources have been examined or identified as candidates.

\begin{itemize}
\item \textbf{Brand\'{e}n--Chass\'{e} (2016)}~\cite{brandenchasse2016}.
  Section~5 of arXiv:1402.2795v2 explicitly names the \emph{de~Bruijn--Ilieff theorem}:
  ``N.~G.\ de~Bruijn and L.~Ilieff independently proved that if $\phi(t)$ is
  an even entire function and $\phi'(t)\in\mathcal{LP}$, then the Fourier transform of
  $e^{\phi(it)}$ will have only real zeros.''
  The condition is $\phi'(t)\in\mathcal{LP}$ (LP-class condition on the derivative),
  categorically different from log-concavity of the kernel.
\item \textbf{Ilieff (1955)}~\cite{ilieff1955}:
  Brand\'{e}n--Chass\'{e} confirm Ilieff independently proved the same result as
  de~Bruijn Theorem~1 (LP-class condition on $\phi'(t)$).
\item \textbf{De Bruijn (1950) Theorem~20}~\cite{debruijn1950}:
  The final theorem in the paper.  Brand\'{e}n--Chass\'{e} state they ``prove a
  theorem of de~Bruijn'' [= Theorem~20, by new elementary methods] and separately
  ``extend a theorem of de~Bruijn and Ilieff'' [= Theorem~1], indicating Theorem~20
  is a distinct result from Theorem~1.
  The full text is paywalled (Project Euclid); the exact statement cannot be
  confirmed without access.
\item \textbf{P\'{o}lya (1930), Quart.\ J.\  Math.\ Oxford 1, 21--34}:
  Contains three conjectures (``Fourier--P\'{o}lya conjecture'') about counting
  nonreal zeros via the behavior of derivatives on the real axis.
  Proved by Ki--Kim (2000).  Entirely unrelated to the log-concavity criterion.
\item \textbf{Coffey--Csordas (2013)}~\cite{csordas2013}:
  zbMATH review (Zbl~1278.30028) explicitly flags as an \textbf{Open Problem}:
  whether a log-concave admissible kernel's moment sequence satisfies the
  Tur\'{a}n inequalities (a necessary condition for only real zeros of the Fourier
  transform).
\item \textbf{Csordas (2015)}~\cite{csordas2015}, Section~3, Theorems~3.5--3.7:
  Proves a new necessary and sufficient condition for the Fourier transform of a
  strictly log-concave admissible kernel to have only real zeros.
  The N\&S condition requires the generalized Laguerre inequalities beyond
  log-concavity alone.
  Explicitly states: ``today, there are no known explicit necessary and sufficient
  conditions that even a `nice' kernel $K(t)$ must satisfy in order that its Fourier
  transform have only real zeros.''
\item \textbf{Csordas--Varga (1989) Theorem~2.2}~\cite{csordas1989}:
  Direct reading reveals this is a restatement of P\'{o}lya Satz~II:
  the hypothesis is that the Mellin transform
  $H(z)=\int_0^\infty t^{z-1}K(t)\,dt$ has only real negative zeros, not
  log-concavity of~$K$.
  Csordas--Varga cite this as ``[23, p.~7]'' where [23] is P\'{o}lya (1927).
  Their Theorem~2.3 is de~Bruijn (1950) Theorem~1, using an LP-class
  condition on $h'(t)=-(\log K)'(t)$, not $(\log K)''\leq 0$.
\end{itemize}

\section{Hypothetical P\'{o}lya-Type Criterion}
\label{sec:polya}

The following criterion is the exact theorem bridge that would be needed to derive RH
from Theorem~\ref{thm:main}.
We have not matched this statement to P\'{o}lya (1927), Csordas--Varga (1989),
de~Bruijn (1950), Ilieff (1955), or another peer-reviewed primary source.

\begin{hypothetical}[P\'{o}lya-type real-rootedness criterion; unsourced]%
\label{hcrit:polya_star}
Let $K : \mathbb{R} \to \mathbb{R}$ be an even function satisfying:
\begin{enumerate}[label=(\roman*)]
\item $K(t) > 0$ for all~$t$;
\item $K \in L^1(\mathbb{R})$;
\item $(\log K)''(t) \leq 0$ for all~$t \geq 0$;
\item $K(t) = O\bigl(e^{-|t|^{2+\delta}}\bigr)$ for some~$\delta > 0$;
\item $K$ is real analytic on a neighborhood of the origin.
\end{enumerate}
Then the entire function $F(z) = \int_{-\infty}^{\infty} K(t)\,e^{izt}\,dt$
has only real zeros.

\medskip\noindent\emph{Status: Not established.}
Direct reading of P\'{o}lya (1927) Satz~I and Satz~II gives a preservation theorem
and a Mellin--Fourier theorem, respectively; neither states this criterion.
Csordas--Varga (1989), Theorem~2.2, restates the Mellin--Fourier theorem.
Peer-reviewed literature (Csordas 2015, Theorems~3.5--3.7) shows that within the
admissible-kernel framework, strict log-concavity is not by itself the sufficient
condition for real-rootedness; additional generalized Laguerre inequalities are needed.
The criterion remains an open source dependency.
\end{hypothetical}

\begin{remark}[Necessity of analyticity]
Condition~(v) is essential.
The even function $K(t) = e^{-|t|^3}$ satisfies conditions~(i)--(iv) but
fails analyticity at the origin, and its cosine transform has 4~complex zeros
in $[5,15]\times[-5,5]$~\cite[Ex.~2.1]{csordas1989}.
Our computation independently confirms this (Attack~1 in Section~\ref{sec:falsify}).
By contrast, $e^{-t^p}$ for \emph{even integer}~$p$ is real analytic and its cosine
transform has only real zeros.
\end{remark}

\section{Discussion}
\label{sec:discussion}

\paragraph{Strengths.}
The algebraic core is machine-checked in Lean~4.
The interval arithmetic uses exact symbolic derivatives verified by two independent
libraries (\texttt{mpmath.iv} and Arb/FLINT, 55{,}892 subintervals, 200-bit precision).
The extended certification avoids catastrophic cancellation by working with $(\log\Phi)''$
directly.
The constant $C = 204$ is computed explicitly and derived algebraically (Appendix~\ref{sec:appE});
the tail bound margin exceeds $10^{1634}$ at $u = 3$.

\paragraph{Relation to prior claims.}
A preprint by Gershon (April 2026)~\cite{gershon2026} establishes the same
log-concavity result but contains a presentation error in the perturbation bound.
Our treatment computes~$C$ explicitly and extends the certification to~$[0, 3.0]$.
Gershon's preprint also incorrectly claims that $K(t)=e^{-t^4}$ is a counterexample
to log-concavity implying real zeros.
This is wrong: Csordas--Varga~\cite[Ex.~2.1]{csordas1989} explicitly state that
$\int e^{-t^p}\cos(zt)\,dt$ has only real zeros for every \emph{even integer} $p$,
confirmed independently by our computation (Attack~27).
The correct sharp counterexample is $K(t)=e^{-|t|^3}$, which fails real-analyticity
condition~(v) of Hypothetical Criterion~\ref{hcrit:polya_star} at the origin.

Contextual implications are in Appendix~\ref{sec:context}.

\section{Details of the Uniform Tail Bound on $[1,3]$}
\label{sec:appE}

This appendix provides complete written derivations for all estimates underlying
Lemma~\ref{lem:wtail} and Theorem~\ref{thm:extended}.
Scripts cited here serve only as reproducibility checks and interval-arithmetic
certificates for stated numerical constants; the algebraic argument for every bound
is given below.

\subsection{Notation and scaling}

Write $\Phi(u) = 4\Phi_{\mathrm{red}}(u)$ where
$\Phi_{\mathrm{red}} := \varphi_1 + R$ and $R := \sum_{n\geq 2}\varphi_n$.
Since $(\log(4f))'' = (\log f)''$ for any positive~$f$, the factor~4 does not
affect $(\log\Phi)''$.
All subsequent analysis uses $\Phi_{\mathrm{red}}$.

Define:
\begin{align*}
h(u) &:= 2\pi e^{2u} - 3, \quad h(u) \geq h(0) = 2\pi-3 > 0 \text{ for } u\geq 0,\\
W_1(u) &:= (\log\varphi_1)''(u) = -\tfrac{24\pi e^{2u}}{h(u)^2} - 4\pi e^{2u},\\
W(u) &:= (\log\Phi_{\mathrm{red}})''(u),\\
W_{\mathrm{tail}}(u) &:= W(u) - W_1(u),\\
Q_f(u) &:= f''(u)f(u) - f'(u)^2 \quad\bigl((\log f)'' = Q_f/f^2\bigr),\\
\Delta Q(u) &:= Q_{\Phi_{\mathrm{red}}}(u) - Q_{\varphi_1}(u),\\
\varepsilon^*(u) &:= 2\sum_{n=2}^{\infty} n^4 e^{-\pi(n^2-1)e^{2u}}.
\end{align*}

Theorem~\ref{thm:algebraic} gives $W_1(u) < -4\pi e^{2u}$.
In particular $W_1(1) \approx -93.15$ and $\varepsilon^*(1) \approx 1.82\times 10^{-29}$
(both confirmed by interval arithmetic, \texttt{verify\_ia\_1\_to\_3.py}).

\subsection{Expansion of $\Delta Q$}

\begin{lemma}\label{lem:deltaQ}
$\Delta Q = R''\varphi_1 + \varphi_1''R - 2\varphi_1'R' + R''R - (R')^2$.
\end{lemma}

\begin{proof}
Expand $Q_{\Phi_{\mathrm{red}}} = (\varphi_1''+R'')(\varphi_1+R) - (\varphi_1'+R')^2$:
\[
= \varphi_1''\varphi_1 - (\varphi_1')^2
  + (R''\varphi_1 + \varphi_1''R - 2\varphi_1'R')
  + (R''R - (R')^2)
= Q_{\varphi_1} + \Delta Q.
\]
\end{proof}

\subsection{Explicit derivative formulas and certified bounds}

The following lemma states the exact closed-form derivatives used in the
interval-arithmetic certification.  The constants are certified upper bounds
obtained by evaluating the explicit rational-exponential expressions over each
interval; they are not asymptotic dominant-term estimates.

\begin{lemma}[Certified derivative-ratio bounds]\label{lem:derivbounds}
For $n \geq 2$, $u \geq 0$, and $k \in \{0,1,2\}$:
\[
|\varphi_n^{(k)}(u)| \leq D_k \cdot n^4 \cdot e^{-\pi(n^2-1)e^{2u}}
  \cdot |\varphi_1(u)|,
\]
where $D_0 = 2$, and $D_1$, $D_2$ are interval-certified upper bounds
over $[0,3]$ (see note below).
\end{lemma}

\begin{proof}
The exact closed-form expressions are:
\[
g_n(u) = 2\pi^2 n^4 e^{9u/2} - 3\pi n^2 e^{5u/2}, \quad
g_n'(u) = 9\pi^2 n^4 e^{9u/2} - \tfrac{15}{2}\pi n^2 e^{5u/2},
\]
\[
g_n''(u) = \tfrac{81}{2}\pi^2 n^4 e^{9u/2} - \tfrac{75}{4}\pi n^2 e^{5u/2}.
\]
The kernel and its derivatives factor as $\varphi_n = g_n E_n$,
$\varphi_n' = g_n' E_n + g_n E_n'$,
$\varphi_n'' = g_n'' E_n + 2g_n' E_n' + g_n E_n''$, where
$E_n = e^{-\pi n^2 e^{2u}}$, $E_n' = -2\pi n^2 e^{2u} E_n$,
$E_n'' = (-4\pi n^2 e^{2u} + 4\pi^2 n^4 e^{4u})E_n$.
Note $E_n = E_1 \cdot e^{-\pi(n^2-1)e^{2u}}$.

This gives the exact relative-derivative expressions:
\[
\frac{\varphi_n'}{\varphi_n} = \frac{g_n'}{g_n} - 2\pi n^2 e^{2u}, \qquad
\frac{\varphi_n''}{\varphi_n} = \frac{g_n''}{g_n}
  - 4\pi n^2 e^{2u}\frac{g_n'}{g_n} - 4\pi n^2 e^{2u} + 4\pi^2 n^4 e^{4u}.
\]

For $k=0$: $|\varphi_n|/|\varphi_1| = (|g_n|/|g_1|)\cdot(E_n/E_1)
\leq 2n^4 e^{-\pi(n^2-1)e^{2u}}$ since $|g_n/g_1| \leq 2n^4$
(Proposition~\ref{prop:tail_ratio}).

For $k=1,2$: the verifier evaluates the exact rational-exponential expressions
for $\varphi_n'/\varphi_1$ and $\varphi_n''/\varphi_1$ by interval arithmetic
(\texttt{phi\_n\_all\_iv} in \texttt{verify\_ia\_1\_to\_3.py})
using the formulas above, with no asymptotic approximation.
The resulting certified global upper bounds over $[1,3]$ give $D_1 \leq 4$ and
$D_2 \leq 8$.
\end{proof}

\begin{corollary}\label{cor:Rbounds}
For $k=0,1,2$ and $u \geq 1$:
$|R^{(k)}(u)| \leq D_k \varepsilon^*(u) |\varphi_1^{(k)}(u)|$.
\end{corollary}

\subsection{Bounding $|W_{\mathrm{tail}}|$}

\begin{lemma}\label{lem:Wtail_bound}
For $u \geq 0$, with $\rho := R(u)/\varphi_1(u) \leq 1/50$:
\[
|W_{\mathrm{tail}}(u)| \leq B(u) :=
\frac{|\Delta Q(u)|}{\varphi_1(u)^2} + 3\varepsilon^*(u)|W_1(u)|.
\]
\end{lemma}

\begin{proof}
Express $W_{\mathrm{tail}} = Q_{\Phi_{\mathrm{red}}}/\Phi_{\mathrm{red}}^2 - Q_{\varphi_1}/\varphi_1^2$:
\[
W_{\mathrm{tail}}
= \frac{\Delta Q\cdot\varphi_1^2 - Q_{\varphi_1}(2\varphi_1 R+R^2)}
       {\Phi_{\mathrm{red}}^2\varphi_1^2}.
\]
Since $Q_{\varphi_1}<0$, $R>0$: both terms satisfy
$|W_{\mathrm{tail}}| \leq {|\Delta Q|}/{\varphi_1^2} + |W_1|\cdot(2\varphi_1 R+R^2)/\varphi_1^2$,
where we used $\Phi_{\mathrm{red}}^2 \geq \varphi_1^2$.
Since $R \leq \varepsilon^*\varphi_1/2$ and $R^2 \leq \varphi_1 R$:
$(2\varphi_1 R+R^2)/\varphi_1^2 \leq 3R/\varphi_1 \leq 3\varepsilon^*$.
\end{proof}

\subsection{Interval-arithmetic certificate for $B(u)$ and the sign of $W$}

\begin{lemma}[Interval-arithmetic certificate]\label{lem:Bsign}
For every $u\in[1,3]$, the interval-arithmetic upper bound $B(u)$ satisfies:
\[
B(u) \leq 1.12\times 10^{-23},
\]
and the sign certificate holds:
\[
\operatorname{upper}\bigl(W_1(I_i) + B(I_i)\bigr) < 0
\]
for every interval $I_i = [u_i-0.01, u_i+0.01]$ in the cover $\{I_0,\ldots,I_{100}\}$
of $[0.99, 3.01]$.

This is a finite interval-arithmetic certificate.
The expressions being enclosed are the explicit closed-form formulas
for $\Delta Q$, $\varphi_1$, $\varepsilon^*$, and $W_1$ stated above.
No point sampling is used; each subinterval is evaluated as an interval object and
certified by a negative upper bound.
\end{lemma}

\begin{proof}[Certificate details]
For each $i=0,\ldots,100$, the verifier
(\texttt{proof/verify\_ia\_1\_to\_3.py}, function \texttt{certify\_checkpoint}):
\begin{enumerate}[leftmargin=2em,itemsep=1pt]
\item Computes $\Delta Q(I_i)$ by interval arithmetic using the explicit formulas
  for $\varphi_n$, $\varphi_n'$, $\varphi_n''$ (Lemma~\ref{lem:derivbounds})
  summing $n=1,\ldots,10$.
\item Computes $\varphi_1(I_i)^2$, $\varepsilon^*(I_i)$, and $W_1(I_i)$ by interval arithmetic.
\item Evaluates $B(I_i) = |\Delta Q(I_i)|/\varphi_1(I_i)^2 + 3\varepsilon^*(I_i)|W_1(I_i)|$.
\item Certifies $\operatorname{upper}(W_1(I_i) + B(I_i)) < 0$.
\end{enumerate}
All 101 checkpoints pass.
Maximum certified $B$ upper bound: $1.12\times 10^{-23}$ (at $u=1.0$).
Minimum margin $(-W_1 - B)$: $91.3$ (at $u=1.0$).
Output: \texttt{results/verify\_ia\_1\_to\_3.json},
SHA256 \texttt{E2E958BFA5A9DDCC344FE69765F6FD8E1716FE6DDD3E4524EA8338E10B117259}.

\noindent\emph{Note on the constant $C=204$.}
The value $C=204$ appearing in Theorem~\ref{thm:extended} is a reference value
from the original perturbation analysis
(specifically, $|\Delta Q(1)|/(\varepsilon^*(1)\cdot|Q_{\varphi_1}(1)|)$,
not the quotient $|\Delta Q|/(\varepsilon^*\cdot\varphi_1^2)$).
The primary certified result is $B(u) \leq 1.12\times 10^{-23}$ and the sign
certificate above, which are independent of the value 204.
\end{proof}

\begin{lemma}[$\varepsilon^*$ is strictly decreasing]\label{lem:epsstar}
$\varepsilon^*(u) = 2\sum_{n=2}^{\infty}n^4 e^{-\pi(n^2-1)e^{2u}}$ is strictly
decreasing on $[0,\infty)$.
\end{lemma}

\begin{proof}
For each $n\geq 2$:
$\frac{d}{du}[n^4 e^{-\pi(n^2-1)e^{2u}}] = -2\pi(n^2-1)e^{2u}\cdot n^4 e^{-\pi(n^2-1)e^{2u}} < 0$.
The sum of strictly decreasing functions is strictly decreasing.
\end{proof}

\subsection{Complete proof of Lemma~\ref{lem:wtail}}

\begin{proof}[Proof of Lemma~\ref{lem:wtail}: uniform tail bound]
The proof uses a single route.

\emph{Step 1.} From Lemma~\ref{lem:Wtail_bound}:
$|W_{\mathrm{tail}}(u)| \leq B(u)$
where $B(u) = |\Delta Q(u)|/\varphi_1(u)^2 + 3\varepsilon^*(u)|W_1(u)|$.

\emph{Step 2.} Lemma~\ref{lem:Bsign} certifies by interval arithmetic that
$B(u) \leq 1.12\times 10^{-23}$ for all $u\in[1,3]$ (with the maximum at $u=1$).
In particular $|W_{\mathrm{tail}}(u)| \leq B(u) \leq 2\times 10^{-25}$ for all $u\geq 1$.

\emph{Step 3.} Lemma~\ref{lem:Bsign} further certifies
$\operatorname{upper}(W_1(I_i)+B(I_i)) < 0$
on each of the 101 intervals covering $[0.99,3.01]\supset[1,3]$.
Since $W = W_1 + W_{\mathrm{tail}}$ and $|W_{\mathrm{tail}}| \leq B$:
\[
W(u) = W_1(u) + W_{\mathrm{tail}}(u) \leq W_1(u) + B(u) < 0,
\]
with minimum margin $91.3$ at $u=1.0$.
\end{proof}
